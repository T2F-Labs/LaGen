% Define cover2 specific commands
\newcommand{\coverrule}{%
    \begin{center}
    \begin{tikzpicture}
        \draw[accent, line width=2pt] (0,0) -- (4,0);
        \draw[lightaccent, line width=1pt] (4.5,0) -- (6,0);
        \draw[border, line width=0.5pt] (6.5,0) -- (7.5,0);

    \end{center}
}
