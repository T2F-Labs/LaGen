\begin{enhancedlist}
    \item \textcolor{brandPrimary}{\textbf{Primary Color}} - Use for main headings, primary buttons, and key brand elements
    \item \textcolor{brandSecondary}{\textbf{Secondary Color}} - Apply to subheadings, secondary buttons, and decorative elements
    \item \textcolor{brandAccent}{\textbf{Accent Color}} - Reserve for success messages, highlights, and call-to-action elements
