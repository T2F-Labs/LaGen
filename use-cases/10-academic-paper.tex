\section{Theoretical Framework}

\begin{theorem}[Quantum Superposition]
A quantum system can exist in a superposition of multiple states until measurement collapses the wavefunction:
\begin{equation}
|\psi\rangle = \alpha|0\rangle + \beta|1\rangle
\end{equation}
where $|\alpha|^2 + |\beta|^2 = 1$.
\end{theorem}

\begin{proof}
This follows directly from the linearity of quantum mechanics and the normalization requirement for probability amplitudes.
\end{proof}

\begin{algorithm}
\caption{Grover's Search Algorithm}
\begin{algorithmic}[1]
\Procedure{GroverSearch}{$N, f$}
    \State Initialize $|\psi\rangle = \frac{1}{\sqrt{N}}\sum_{x=0}^{N-1}|x\rangle$
    \State $iterations \gets \lfloor\frac{\pi}{4}\sqrt{N}\rfloor$
    \For{$i = 1$ to $iterations$}
        \State Apply oracle $U_f$
        \State Apply diffusion operator $U_s$
    \EndFor
    \State Measure and return result
\EndProcedure
\end{algorithmic}
\end{algorithm}
