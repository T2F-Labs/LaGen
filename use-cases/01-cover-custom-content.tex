\usepackage{eso-pic}
\usepackage{tikz}

% Custom cover page command
\newcommand{\customCover}[5]{%
  % #1: Main Title
  % #2: Subtitle  
  % #3: Author
  % #4: Organization
  % #5: Date
  \thispagestyle{empty}
  \begin{center}
    \vspace*{2cm}
    
    % Custom title design
    {\Huge\bfseries\color{primary}#1}\\[1cm]
    {\LARGE\color{secondary}#2}\\[2cm]
    
    % Custom decorative element
    \begin{tikzpicture}
      \draw[primary, line width=2pt] (-3,0) -- (3,0);
      \fill[accent] (0,0) circle (0.1cm);
    \end{tikzpicture}
    
    \vspace{2cm}
    
    % Author and organization info
    {\large\color{primary}\textbf{Author:} #3}\\[0.5cm]
    {\large\color{primary}\textbf{Organization:} #4}\\[0.5cm]
    {\large\color{primary}\textbf{Date:} #5}
    
    \vfill
    
    % Custom footer design
    \begin{tikzpicture}
      \draw[secondary, line width=1pt] (-4,0) rectangle (4,1);
      \node[color=primary] at (0,0.5) {\textbf{Custom Document Design}};
    \end{tikzpicture}
  \end{center}
  \newpage
}

% ========== DOCUMENT BEGIN ==========
\begin{document}

% Use custom cover
\customCover{%
  LaTeX Modular System%           Title
}{%
  Advanced Document Architecture% Subtitle  
}{%
  John Doe%                      Author
}{%
  Tech Solutions Inc%            Organization
}{%
  November 2024%                 Date
}
